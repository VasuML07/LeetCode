🔥 Sliding Window

Core Intuition

Expand → Break → Shrink → Repeat

[2 1 5 1 3 2] [-----] L R
Expand → Check → Shrink → Update → Repeat

Example

[2,1]=3 [1,5]=6 [5,1]=6

Java

public int maxSum(int[] arr, int k) { int sum = 0; for(int i=0;i<k;i++) sum+=arr[i]; int max=sum; for(int i=k;i<arr.length;i++){ sum+=arr[i]-arr[i-k]; max=Math.max(max,sum); } return max; }

🔥 Two Pointers

Intuition

Two pointers move based on condition

1 2 3 4 5 6 7 8 L R
public int[] twoSum(int[] arr, int target){ int l=0,r=arr.length-1; while(l<r){ int sum=arr[l]+arr[r]; if(sum==target) return new int[]{l,r}; else if(sum<target) l++; else r--; } return new int[]{-1,-1}; }

🔥 Binary Search

1 3 5 7 9 ↑ mid
mid → compare → discard half
public int bs(int[] arr,int t){ int l=0,r=arr.length-1; while(l<=r){ int m=l+(r-l)/2; if(arr[m]==t) return m; else if(arr[m]<t) l=m+1; else r=m-1; } return -1; }

🔥 DFS

A → B → D → back → C
void dfs(int node, boolean[] vis, List<List<Integer>> g){ vis[node]=true; for(int nei:g.get(node)){ if(!vis[nei]) dfs(nei,vis,g); } }

🔥 BFS

A → B → C → D → E
void bfs(int s,List<List<Integer>> g){ Queue<Integer> q=new LinkedList<>(); boolean[] vis=new boolean[g.size()]; q.add(s); vis[s]=true; while(!q.isEmpty()){ int node=q.poll(); for(int nei:g.get(node)){ if(!vis[nei]){ vis[nei]=true; q.add(nei); } } } }

🔥 Dynamic Programming

dp[i] = dp[i-1] + dp[i-2]
int fib(int n){ int[] dp=new int[n+1]; dp[1]=1; for(int i=2;i<=n;i++) dp[i]=dp[i-1]+dp[i-2]; return dp[n]; }

🔥 Backtracking

[] → [1] → [1,2] → backtrack
void bt(List<Integer> path,int[] nums,boolean[] used){ if(path.size()==nums.length){ System.out.println(path); return; } for(int i=0;i<nums.length;i++){ if(used[i]) continue; used[i]=true; path.add(nums[i]); bt(path,nums,used); path.remove(path.size()-1); used[i]=false; } }
